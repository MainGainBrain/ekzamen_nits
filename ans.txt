FROM security_journal

SELECT *
FROM pc inner join applications on pc.id = applications.source_pc
where vlan_pc = 460 and application = 'Mozila Firefox' and session_time like '%17:25%'

select *
from employees 
where pc in (SELECT pc.id
FROM pc inner join applications on pc.id = applications.source_pc
where vlan_pc = 460 and application = 'Mozila Firefox' and session_time like '%17:25%') and status = 13

select name, surname, pc, report
from employees inner join employee_report on employees.uid = employee_report.uid_employee
where (pc in (SELECT pc.id
FROM pc inner join applications on pc.id = applications.source_pc
where vlan_pc = 460 and application = 'Mozila Firefox' and session_time like '%17:25%') and status = 13) or name IN ('Август', 'Алиса', 'Натан','Диана');
